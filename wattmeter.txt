const int sensorPin = A0;
const float VOLTAGE = 230.0; 

// Коэффициент под твои резисторы (320 Ом)
// Если 300Вт пила показывает мало, увеличь это число (например, до 13.0)
const float CALIB = 12.2; 

void setup() {
  Serial.begin(9600);
  // Нам больше не нужно инициализировать дисплей
}

void loop() {
  double sumSq = 0;
  int samples = 1000;

  for (int i = 0; i < samples; i++) {
    int raw = analogRead(sensorPin);
    // 512 - это центр твоего делителя (1к + 1к)
    float vRaw = (raw - 512.0) * (5.0 / 1024.0); 
    sumSq += (double)vRaw * vRaw;
    delayMicroseconds(100);
  }

  double vRMS = sqrt(sumSq / samples);
  double currentRMS = vRMS * CALIB; 
  double watts = currentRMS * VOLTAGE;

  // ЖЕСТКАЯ ОЧИСТКА МУСОРА
  // Если на пустом проводе было 56, то отсечка на 60 сделает там честный 0
  if (watts < 60.0) watts = 0; 

  // Шлем только чистое число для графика
  Serial.println(watts);
  
  delay(100); // Теперь обновляем график чаще (10 раз в секунду!)
}
