    <script>
        let chart;
        let buffer = "";

        async function connectSerial() {
            try {
                const port = await navigator.serial.requestPort();
                await port.open({ baudRate: 9600 });
                const reader = port.readable.getReader();
                
                initChart();

                while (true) {
                    const { value, done } = await reader.read();
                    if (done) break;
                    buffer += new TextDecoder().decode(value);
                    
                    if (buffer.includes('\n')) {
                        let lines = buffer.split('\n');
                        let lastValidData = lines[lines.length - 2];
                        buffer = lines[lines.length - 1];

                        let watts = parseFloat(lastValidData);
                        if (!isNaN(watts)) {
                            updateUI(watts);
                        }
                    }
                }
            } catch (e) {
                alert("Ошибка: " + e.message);
            }
        }

        function initChart() {
            const ctx = document.getElementById('myChart').getContext('2d');
            chart = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: [],
                    datasets: [{
                        label: 'Мощность (W)',
                        data: [],
                        borderColor: '#00ff00',
                        backgroundColor: 'rgba(0, 255, 0, 0.1)',
                        borderWidth: 2,
                        tension: 0.3,
                        pointRadius: 0
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        zoom: {
                            pan: { enabled: true, mode: 'x', threshold: 5 },
                            zoom: {
                                wheel: { enabled: true, modifierKey: 'ctrl' },
                                mode: 'x'
                            }
                        },
                        legend: { display: false }
                    },
                    scales: {
                        x: { grid: { color: '#444' }, ticks: { color: '#888' } },
                        y: { grid: { color: '#444' }, ticks: { color: '#888' }, beginAtZero: true, suggestedMax: 500 }
                    }
                }
            });
        }

        function updateUI(val) {
            document.getElementById('val').innerText = val.toFixed(1) + " W";
            const now = new Date().toLocaleTimeString();
            
            chart.data.labels.push(now);
            chart.data.datasets[0].data.push(val);

            // Ограничиваем количество точек до 100, чтобы не тормозило
            if (chart.data.labels.length > 100) {
                chart.data.labels.shift();
                chart.data.datasets[0].data.shift();
            }
            chart.update('none'); // Быстрое обновление без анимации
        }
    </script>
